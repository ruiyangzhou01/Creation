\section{微分方程理论：平衡点的性质}

\subsection{无病平衡点与稳定性分析} \label{subsec:dfep}

在传染病动力学中，无病平衡点是指在人群中没有任何感染的状态下，传染病的传播达到的一个平衡状态。

\begin{theorem}[无病平衡点的存在唯一性]
	\label{thm:dfe_eut}
	模型~\eqref{eq:model}的无病平衡点 $X^0$ 存在且唯一，其值为：
	\begin{equation*}
		\begin{aligned}
			S_1^0 & = \dfrac{\Lambda}{\nu_1+\mu_1 +g },                                                \\
			S_2^0 & = \dfrac{g}{\nu_2+\mu_{2}}S_1^0,                                                     \\
			V_1^0 & = \dfrac{\Lambda \,\nu_1 }{{\left(\mu_1+g \right)}\,{\left(\nu_1+\mu_1 +g \right)}}, \\
			V_2^0 & = \dfrac{g\,{\left(g\,\nu_2 + \nu_2\,\mu_1 + \nu_1\,\mu_2 +\nu_1 \,\nu_2 \right)}} {\nu_1 \mu_2 \,{\left(\nu_2+\mu_2 \right)}}V_1^0.
		\end{aligned}
	\end{equation*}
\end{theorem}

\begin{theorem}[无病平衡点的全局稳定性]
	\label{thm:dfe_global_stability}
	假设下列条件成立时，
	\begin{equation*}
		\label{eq:DFE_coefficients_condition}
		\begin{aligned}
			& A_1 D_1 -B_{11} C_1 >0, \quad A_2 D_2 -B_{22} C_2 >0, \\
			& -A_1 A_2 D_1 D_2+\left(B_{11} B_{22} -B_{12} B_{21}\right) C_1 C_2>0.
		\end{aligned}
	\end{equation*}
	则无病平衡点 $X^0$ 是全局渐近稳定的。
\end{theorem}

\subsection{地方病平衡点与稳定性分析} \label{subsec:eep}

传染病动力学中，地方病平衡点是指在特定的地区和特定的时间内，一个传染病的感染人口数量不再上升或下降，而是保持在一个稳定的水平上。在这种情况下，疾病传播呈现出一种平衡状态，称为地方病平衡点。

\begin{theorem}[地方病平衡点的存在性和唯一性]
	\label{thm:eep}
	如果以下条件之一成立，
	\begin{description}
		\item[条件\romannum{1}.]
		$\begin{aligned}[t]
			 & a_{11} \neq 0,\ b_{11} \neq 0,
			 \\
			 & a_{11} \left( a_1 b_{12} - a_{12} b_1 \right) ^2 = 2 a_2 \left( a_{11} b_{12} - a_{12} b_{11} \right) \left(a_1 b_{11}-a_{11} b_1\right),
			 \\
			 & b_{11} \left(a_1 b_{12}-a_{12} b_1\right)^2 \\
			 & = b_{22} \left( a_1 b_{11}-a_{11} b_1 \right) ^2 +2 b_2 \left(a_1 b_{12}-a_{12} b_1\right) \left(a_1 b_{11}-a_{11} b_1\right);
		\end{aligned}$
		\item[条件\romannum{2}.]
		$\begin{aligned}[t]
			 & a_{11} = 0,\ b_{11} \neq 0,\ a_{12} b_{12} \neq 0,
			 \\
			 & \left( a_{12} b_1+a_1 b_{12} - a_2 b_{11} \right)^2-4 a_{12} b_{12} a_1 b_1 =0,
			 \\
			 & \quad - {b_{12} }^2 \,{{\left(a_2 \,b_{11} -a_1 \,b_{12} +a_{12} \,b_1 \right)}}^2 / \left[ b_{11} {\left(a_1 \,b_{12} -a_2 \,b_{11} +a_{12} \,b_1 \right)}\right]
			 \\
			 & = 4\,a_{12} \,b_2 \,b_{12} -a_1 \,b_{12} \,b_{22} +a_2 \,b_{11} \,b_{22} -a_{12} \,b_1 \,b_{22};
		\end{aligned}$
		\item[条件\romannum{3}.]
		$\begin{aligned}[t]
			 & a_{11} = 0, \ b_{11} \neq 0,\ a_{12} b_{12} = 0,
			 \\
			 & \quad b_{11} \,{\left(b_{22} \,{a_1 }^2 \,b_1 +2\,a_2 \,b_2 \,b_{11} \,a_1 +2\,a_2 \,a_{12} \,{b_1 }^2 \right)}
			 \\
			 & = 2 a_1 b_2 b_{11}(\, \,b_{12} \,a_1 + \, \,a_{12} \,b_1 ) +{a_2 }^2 \,b_1 \,{b_{11} }^2 +{a_{12} }^2 \,{b_1 }^3;
		\end{aligned}$
		\item[条件\romannum{4}.]
		$\begin{aligned}[t]
			 & a_{11} \neq 0,\ b_{11} = 0,\ a_{11} b_{22}-2a_{12} b_{12} \neq 0,
			 \\
			 & {{\left({a_{12} }^2 \,b_1 -a_1 \,a_{12} \,b_{12} -a_{11} \,a_{12} \,b_2 +a_1 \,a_{11} \,b_{22} \right)}}^2
			 \\
			 & +2\,a_2 \,a_{11} \,{\left(2\,a_{12} \,b_{12} -a_{11} \,b_{22} \right)}\,{\left(a_1 \,b_{12} -a_{11} \,b_2 +a_{12} \,b_1 \right)}=0,
			 \\
			 & (a_{11} b_2-a_1 b_{12}-a_{12} b_1 )^2 +2 a_1 b_1 (a_{11} b_{22}-2a_{12} b_{12}) =0;
			 \\
		\end{aligned}$
	\end{description}
	那么模型~\eqref{eq:model} 存在一个唯一的地方病平衡点 $ X^* $.
\end{theorem}

\begin{theorem}[地方病平衡点的全局稳定性]
	\label{thm:eep_stability}
	假设下列条件成立时，
	\begin{equation*}
		\label{eq:EEP_constraint-simplified}
		\begin{aligned}
			& m_{12}m_{13}m_{21}m_{44}-m_{11}m_{14}m_{22}m_{53} >0, \\[+0.5ex]
			& m_{11}m_{22}-m_{12}m_{21}>0, \\[+0.5ex]
			& m_{13}m_{44}-m_{14}m_{53}>0, \\[+0.5ex]
			& m_{33}m_{44}-m_{11}m_{53}>0.
		\end{aligned}
	\end{equation*}
	则地方病平衡点 $X^*$ 是全局渐近稳定的。
\end{theorem}


\section{神经网络模型：网络训练与预测}

本节将详细介绍使用LSTM网络模型来预测 COVID-19 的动力学特征的过程。
