			\subblock{Hamiltonian for $N_q$ qubits}{%
				\begin{equation*}
					H = J \sum_{j=1}^{N_q-1} \sigma^x_{j} \sigma^x_{j+1} + \sum_{j=1}^{N_q} h_j \sigma^z_{j} \text{,}
				\end{equation*}
			}

			\subblock{Observables for $N_c$ classes}{%
				\vspace*{-1em}
				\begin{equation*}
					O_{\alpha} = \dfrac{1}{N_S} \sum_{k=1}^{N_S} \ket{\Psi_{k}^{(\alpha)} (t)}\bra{\Psi_{k}^{(\alpha)} (t)},
					\alpha = 1, 2, \ldots, N_c.
				\end{equation*}
			}

			\subblock{Two kinds of measurement}{%
				1. \textbf{Exact expectation:}
				$ \langle O_{\alpha}\rangle = \bra{\Psi}O_{\alpha}\ket{\Psi} $.

				2. \textbf{Shadow tomography:}
				$ \langle O_{\alpha}\rangle = \frac{1}{M} \sum_{m=1}^{M} \text{Tr} \Big(\rho^{(m)}O_{\alpha}\Big)$.
			}

			\subblock{Loss function}{%
				\begin{equation*}
					\mathcal{C} {=} \frac{1}{N_{\textrm trn}} \sum_{\{ \mathbf{x_1}, \mathbf{x_2} \}} \Big||\braket{\Psi_1(t) | \Psi_2(t)}|^2 - \delta{y_1, y_2} \Big|.
				\end{equation*}
			}
